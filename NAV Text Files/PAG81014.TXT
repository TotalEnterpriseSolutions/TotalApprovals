    LOCAL PROCEDURE SetBTNState@1000000000();
    VAR
      LrecPSetup@1000000000 : Record 81000;
      LrecDProcType@1000000001 : Record 81001;
      LcduApprovalsMgmt@1000000002 : Codeunit 1535;
      GeneralLedgerSetup@1000000003 : Record 98;
      ProcurementWorkflow@1000000004 : Record 81001;
    BEGIN
      GeneralLedgerSetup.GET;
      GD1Caption := GeneralLedgerSetup."Global Dimension 1 Code";
      GD2Caption := GeneralLedgerSetup."Global Dimension 2 Code";
      LrecPSetup.FINDFIRST;
      CLEAR(GbooEditable);
      IF "Error Text" <> '' THEN BEGIN
        GbooError := TRUE;
      END;
      IF LrecDProcType.GET("Procurement Workflow")THEN BEGIN
        GbooShowGD1 := LrecDProcType."Show Dimension 1";
        GbooShowGD2 := LrecDProcType."Show Dimension 2";
        GbooShowExpDate := LrecDProcType."Due Date Required";
      END;
      CASE Status OF
       Status::Open:
         BEGIN
          GbooDelete:= TRUE;
          GbooApprovalSubmit := TRUE;
          GbooEditable := TRUE;
         END;
       Status::Released:
         BEGIN
          GbooDelete := FALSE;
          DocNoVisible := TRUE;
          GbooPost := TRUE;
          GbooApproved:= TRUE;
          GbooSubmitted := TRUE;
          GbooEditable := TRUE;
         END;
        Status::"Pending Approval":
          BEGIN
          GbooDelete := FALSE;
          GbooSubmitted := TRUE;
          GbooApprovalCancel := TRUE;
          GbooEditable := TRUE;
          END;
      END;
      OpenApprovalEntriesExistForCurrUser := LcduApprovalsMgmt.HasOpenApprovalEntriesForCurrentUser(RECORDID);
      IF OpenApprovalEntriesExistForCurrUser THEN BEGIN
        GbooApprovalCancel := FALSE;
        GbooApprovalSubmit := FALSE;
      END;
    END;
